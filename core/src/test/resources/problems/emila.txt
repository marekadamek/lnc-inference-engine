# endhumans
# enddragon

# Reguły na śmierć
(dragon & attack & !hassword => N(dead))
& (dragon & attack & hassword => N(enddragon))

# 2 pierwsze przyloty smoka
& ((inprison | monks) & !dragon => N(dragon))

# po ataku dragon ucieka
& (dragon => N(!dragon))

# Ucieczka z więzienia
& (inprison & dragon => N(!inprison))

# first annihilation order
& (!inprison & !dragon & !firstao => N(firstao))

# zabij plemie
& (firstao & killtribe => N(secondao))

# nie zabijaj plemienia
& (firstao & !killtribe => N(hassword))

# idz do mnichow
& (secondao | hassword => N(monks))

# 2 atak smoka
& (monks & dragon & !attack => N(village))

# rozgalezienie w wiosce
& (village => N((battle & !dragon) | (!battle & dragon)))

# wybor strony
& (battle & !joinhumans => N(endelves))
& (battle & joinhumans => N(endhumans))
