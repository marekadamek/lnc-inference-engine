#FARMER DILEMMA

# F - farmer
# W - wolf
# S - sheep
# C - cabbage

# B - boat (1 - inside, 0 - outside)
# R - river (1 - right, 0 - left)

# BF, RF - farmer
# BW, RW - wolf
# BS, RS - sheep
# BC, RC - cabbage

C(bf)
& (!bf => C(rf))
& (bf => !C(rf))
& (bw => bf & (rw <=> rf))
& (bw => !C(rw))
& (C(rw) => C(rf) & (rw <=> rf))
& (C(rw) => !bw & C(bw))
& (C(bw) => (bw <=> bf) & (rw <=> rf) & C(bf))
& (bs => bf & (rs <=> rf))
& (bs => !C(rs)) & (C(rs) => C(rf) & (rs <=> rf))
& (C(rs) => !bs & C(bs)) & (C(bs) => (bs <=> bf) & (rs <=> rf) & C(bf))
& (bc => bf & (rc <=> rf))
& (bc => !C(rc)) & (C(rc) => C(rf) & (rc <=> rf))
& (C(rc) => !bc & C(bc)) & (C(bc) => (bc <=> bf) & (rc <=> rf) & C(bf))
& (!bw & !bs & (rw <=> rs) => (rw <=> rf) & !bf)
& (!bs & !bc & (rs <=> rc) => (rs <=> rf) & !bf)
& (bc => !bw & !bs) & (bs => !bw & !bc) & (bw => !bc & !bs)
& N((bc => !bw & !bs) & (bs => !bw & !bc) & (bw => !bc & !bs))
